/*****  Execution to handle failover *****/
#!/usr/bin/env bash
set -uo pipefail  # Omit `-e`

run_step() {
  echo "Running $1"
  if $1; then
    echo "$1 succeeded"
  else
    echo  "$1 failed"
  fi
}

cd "$(dirname "$0")"

run_step "../config/kind/deploy.sh"
sleep 5
run_step "../config/ingress/deploy.sh"
sleep 5
run_step "../config/monitoring/deploy.sh"
sleep 5
run_step "../config/postgres/deploy.sh"
sleep 60
run_step "./populate-db.sh"



apiVersion: v1
kind: Service
metadata:
  name: localdb-nodeport
  namespace: postgres
spec:
  type: NodePort
  selector:
    cnpg.io/cluster: localdb
    role: primary
  ports:
    - port: 5432
      targetPort: 5432
      nodePort: 30007

psql -h localhost -p 6543 -U localuser localdb
